55/// Checks if a row group contains delete operations by examining the min value of op_type column.
56///
57/// Returns `Ok(true)` if the row group contains delete operations, `Ok(false)` if it doesn't,
58/// or an error if the statistics are not present or cannot be decoded.
59pub(crate) fn row_group_contains_delete(
60    parquet_meta: &ParquetMetaData,
61    row_group_index: usize,
62    file_path: &str,
63) -> Result<bool> {
64    let row_group_metadata = &parquet_meta.row_groups()[row_group_index];
65
66    // safety: The last column of SST must be op_type
67    let column_metadata = &row_group_metadata.columns().last().unwrap();
68    let stats = column_metadata
69        .statistics()
70        .context(StatsNotPresentSnafu { file_path })?;
71    stats
72        .min_bytes_opt()
73        .context(StatsNotPresentSnafu { file_path })?
74        .try_into()
75        .map(i32::from_le_bytes)
76        .map(|min_op_type| min_op_type == OpType::Delete as i32)
77        .ok()
78        .context(DecodeStatsSnafu { file_path })
79}

81/// A range of a parquet SST. Now it is a row group.
82/// We can read different file ranges in parallel.
83#[derive(Clone)]
84pub struct FileRange {
85    /// Shared context.
86    context: FileRangeContextRef,
87    /// Index of the row group in the SST.
88    row_group_idx: usize,
89    /// Row selection for the row group. `None` means all rows.
90    row_selection: Option<RowSelection>,
91}
92
93impl FileRange {
94    /// Creates a new [FileRange].
95    pub(crate) fn new(
96        context: FileRangeContextRef,
97        row_group_idx: usize,
98        row_selection: Option<RowSelection>,
99    ) -> Self {
100        Self {
101            context,
102            row_group_idx,
103            row_selection,
104        }
105    }
106
107    /// Returns true if [FileRange] selects all rows in row group.
108    fn select_all(&self) -> bool {
109        let rows_in_group = self
110            .context
111            .reader_builder
112            .parquet_metadata()
113            .row_group(self.row_group_idx)
114            .num_rows();
115
116        let Some(row_selection) = &self.row_selection else {
117            return true;
118        };
119        row_selection.row_count() == rows_in_group as usize
120    }
121
122    /// Performs pruning before reading the [FileRange].
123    /// It use latest dynamic filters with row group statistics to prune the range.
124    ///
125    /// Returns false if the entire range is pruned and can be skipped.
126    fn in_dynamic_filter_range(&self) -> bool {
127        if self.context.base.dyn_filters.is_empty() {
128            return true;
129        }
130        let curr_row_group = self
131            .context
132            .reader_builder
133            .parquet_metadata()
134            .row_group(self.row_group_idx);
135        let read_format = self.context.read_format();
136        let prune_schema = &self.context.base.prune_schema;
137        let stats = RowGroupPruningStats::new(
138            std::slice::from_ref(curr_row_group),
139            read_format,
140            self.context.base.expected_metadata.clone(),
141            self.compute_skip_fields(),
142        );
143
144        // not costly to create a predicate here since dynamic filters are wrapped in Arc
145        let pred = Predicate::new(vec![]).with_dyn_filters(self.context.base.dyn_filters.clone());
146
147        pred.prune_with_stats(&stats, prune_schema.arrow_schema())
148            .first()
149            .cloned()
150            .unwrap_or(true) // unexpected, not skip just in case
151    }
152
153    fn compute_skip_fields(&self) -> bool {
154        match self.context.base.pre_filter_mode {
155            PreFilterMode::All => false,
156            PreFilterMode::SkipFields => true,
157            PreFilterMode::SkipFieldsOnDelete => {
158                // Check if this specific row group contains delete op
159                row_group_contains_delete(
160                    self.context.reader_builder.parquet_metadata(),
161                    self.row_group_idx,
162                    self.context.reader_builder.file_path(),
163                )
164                .unwrap_or(true)
165            }
166        }
167    }
168
169    /// Returns a reader to read the [FileRange].
170    pub(crate) async fn reader(
171        &self,
172        selector: Option<TimeSeriesRowSelector>,
173        fetch_metrics: Option<&ParquetFetchMetrics>,
174    ) -> Result<Option<PruneReader>> {
175        if !self.in_dynamic_filter_range() {
176            return Ok(None);
177        }
178        let parquet_reader = self
179            .context
180            .reader_builder
181            .build(
182                self.row_group_idx,
183                self.row_selection.clone(),
184                fetch_metrics,
185            )
186            .await?;
187
188        let use_last_row_reader = if selector
189            .map(|s| s == TimeSeriesRowSelector::LastRow)
190            .unwrap_or(false)
191        {
192            // Only use LastRowReader if row group does not contain DELETE
193            // and all rows are selected.
194            let put_only = !self
195                .context
196                .contains_delete(self.row_group_idx)
197                .inspect_err(|e| {
198                    error!(e; "Failed to decode min value of op_type, fallback to RowGroupReader");
199                })
200                .unwrap_or(true);
201            put_only && self.select_all()
202        } else {
203            // No selector provided, use RowGroupReader
204            false
205        };
206
207        // Compute skip_fields once for this row group
208        let skip_fields = self.context.should_skip_fields(self.row_group_idx);
209
210        let prune_reader = if use_last_row_reader {
211            // Row group is PUT only, use LastRowReader to skip unnecessary rows.
212            let reader = RowGroupLastRowCachedReader::new(
213                self.file_handle().file_id().file_id(),
214                self.row_group_idx,
215                self.context.reader_builder.cache_strategy().clone(),
216                RowGroupReader::new(self.context.clone(), parquet_reader),
217            );
218            PruneReader::new_with_last_row_reader(self.context.clone(), reader, skip_fields)
219        } else {
220            // Row group contains DELETE, fallback to default reader.
221            PruneReader::new_with_row_group_reader(
222                self.context.clone(),
223                RowGroupReader::new(self.context.clone(), parquet_reader),
224                skip_fields,
225            )
226        };
227
228        Ok(Some(prune_reader))
229    }

231    /// Creates a flat reader that returns RecordBatch.
232    pub(crate) async fn flat_reader(
233        &self,
234        fetch_metrics: Option<&ParquetFetchMetrics>,
235    ) -> Result<Option<FlatPruneReader>> {
236        if !self.in_dynamic_filter_range() {
237            return Ok(None);
238        }
239        let parquet_reader = self
240            .context
241            .reader_builder
242            .build(
243                self.row_group_idx,
244                self.row_selection.clone(),
245                fetch_metrics,
246            )
247            .await?;
248
249        // Compute skip_fields once for this row group
250        let skip_fields = self.context.should_skip_fields(self.row_group_idx);
251
252        let flat_row_group_reader = FlatRowGroupReader::new(self.context.clone(), parquet_reader);
253        let flat_prune_reader = FlatPruneReader::new_with_row_group_reader(
254            self.context.clone(),
255            flat_row_group_reader,
256            skip_fields,
257        );
258
259        Ok(Some(flat_prune_reader))
260    }

337    /// Determines whether to skip field filters based on PreFilterMode and row group delete status.
338    pub(crate) fn should_skip_fields(&self, row_group_idx: usize) -> bool {
339        match self.base.pre_filter_mode {
340            PreFilterMode::All => false,
341            PreFilterMode::SkipFields => true,
342            PreFilterMode::SkipFieldsOnDelete => {
343                // Check if this specific row group contains delete op
344                self.contains_delete(row_group_idx).unwrap_or(true)
345            }
346        }
347    }

353    }

356/// Mode to pre-filter columns in a range.
357#[derive(Debug, Clone, Copy)]
358pub enum PreFilterMode {
359    /// Filters all columns.
360    All,
361    /// If the range doesn't contain delete op or doesn't have statistics, filters all columns.
362    /// Otherwise, skips filtering fields.
363    SkipFieldsOnDelete,
364    /// Always skip fields.
365    SkipFields,
366}

387impl RangeBase {
388    /// TRY THE BEST to perform pushed down predicate precisely on the input batch.
389    /// Return the filtered batch. If the entire batch is filtered out, return None.
390    ///
391    /// Supported filter expr type is defined in [SimpleFilterEvaluator].
392    ///
393    /// When a filter is referencing primary key column, this method will decode
394    /// the primary key and put it into the batch.
395    ///
396    /// # Arguments
397    /// * `input` - The batch to filter
398    /// * `skip_fields` - Whether to skip field filters based on PreFilterMode and row group delete status
399    pub(crate) fn precise_filter(
400        &self,
401        mut input: Batch,
402        skip_fields: bool,
403    ) -> Result<Option<Batch>> {
404        let mut mask = BooleanBuffer::new_set(input.num_rows());
405
406        // Run filter one by one and combine them result
407        // TODO(ruihang): run primary key filter first. It may short circuit other filters
408        for filter_ctx in &self.filters {
409            let filter = match filter_ctx.filter() {
410                MaybeFilter::Filter(f) => f,
411                // Column matches.
412                MaybeFilter::Matched => continue,
413                // Column doesn't match, filter the entire batch.
414                MaybeFilter::Pruned => return Ok(None),
415            };
416            let result = match filter_ctx.semantic_type() {
417                SemanticType::Tag => {
418                    let pk_values = if let Some(pk_values) = input.pk_values() {
419                        pk_values
420                    } else {
421                        input.set_pk_values(
422                            self.codec
423                                .decode(input.primary_key())
424                                .context(DecodeSnafu)?,
425                        );
426                        input.pk_values().unwrap()
427                    };
428                    let pk_value = match pk_values {
429                        CompositeValues::Dense(v) => {
430                            // Safety: this is a primary key
431                            let pk_index = self
432                                .read_format
433                                .metadata()
434                                .primary_key_index(filter_ctx.column_id())
435                                .unwrap();
436                            v[pk_index]
437                                .1
438                                .try_to_scalar_value(filter_ctx.data_type())
439                                .context(DataTypeMismatchSnafu)?
440                        }
441                        CompositeValues::Sparse(v) => {
442                            let v = v.get_or_null(filter_ctx.column_id());
443                            v.try_to_scalar_value(filter_ctx.data_type())
444                                .context(DataTypeMismatchSnafu)?
445                        }
446                    };
447                    if filter
448                        .evaluate_scalar(&pk_value)
449                        .context(RecordBatchSnafu)?
450                    {
451                        continue;
452                    } else {
453                        // PK not match means the entire batch is filtered out.
454                        return Ok(None);
455                    }
456                }
457                SemanticType::Field => {
458                    // Skip field filters if skip_fields is true
459                    if skip_fields {
460                        continue;
461                    }
462                    // Safety: Input is Batch so we are using primary key format.
463                    let Some(field_index) = self
464                        .read_format
465                        .as_primary_key()
466                        .unwrap()
467                        .field_index_by_id(filter_ctx.column_id())
468                    else {
469                        continue;
470                    };
471                    let field_col = &input.fields()[field_index].data;
472                    filter
473                        .evaluate_vector(field_col)
474                        .context(RecordBatchSnafu)?
475                }
476                SemanticType::Timestamp => filter
477                    .evaluate_vector(input.timestamps())
478                    .context(RecordBatchSnafu)?,
479            };
480
481            mask = mask.bitand(&result);
482        }
483
484        input.filter(&BooleanArray::from(mask).into())?;
485
486        Ok(Some(input))
487    }

489    /// Filters the input RecordBatch by the pushed down predicate and returns RecordBatch.
490    ///
491    /// It assumes all necessary tags are already decoded from the primary key.
492    ///
493    /// # Arguments
494    /// * `input` - The RecordBatch to filter
495    /// * `skip_fields` - Whether to skip field filters based on PreFilterMode and row group delete status
496    pub(crate) fn precise_filter_flat(
497        &self,
498        input: RecordBatch,
499        skip_fields: bool,
500    ) -> Result<Option<RecordBatch>> {
501        let mask = self.compute_filter_mask_flat(&input, skip_fields)?;
502
503        // If mask is None, the entire batch is filtered out
504        let Some(mask) = mask else {
505            return Ok(None);
506        };
507
508        let filtered_batch =
509            datatypes::arrow::compute::filter_record_batch(&input, &BooleanArray::from(mask))
510                .context(ComputeArrowSnafu)?;
511
512        if filtered_batch.num_rows() > 0 {
513            Ok(Some(filtered_batch))
514        } else {
515            Ok(None)
516        }
517    }
518
519    /// Computes the filter mask for the input RecordBatch based on pushed down predicates.
520    ///
521    /// Returns `None` if the entire batch is filtered out, otherwise returns the boolean mask.
522    ///
523    /// # Arguments
524    /// * `input` - The RecordBatch to compute mask for
525    /// * `skip_fields` - Whether to skip field filters based on PreFilterMode and row group delete status
526    pub(crate) fn compute_filter_mask_flat(
527        &self,
528        input: &RecordBatch,
529        skip_fields: bool,
530    ) -> Result<Option<BooleanBuffer>> {
531        let mut mask = BooleanBuffer::new_set(input.num_rows());
532
533        let flat_format = self
534            .read_format
535            .as_flat()
536            .context(crate::error::UnexpectedSnafu {
537                reason: "Expected flat format for precise_filter_flat",
538            })?;
539
540        // Decodes primary keys once if we have any tag filters not in projection
541        let mut decoded_pks: Option<DecodedPrimaryKeys> = None;
542        // Cache decoded tag arrays by column id to avoid redundant decoding
543        let mut decoded_tag_cache: HashMap<ColumnId, ArrayRef> = HashMap::new();
544
545        // Run filter one by one and combine them result
546        for filter_ctx in &self.filters {
547            let filter = match filter_ctx.filter() {
548                MaybeFilter::Filter(f) => f,
549                // Column matches.
550                MaybeFilter::Matched => continue,
551                // Column doesn't match, filter the entire batch.
552                MaybeFilter::Pruned => return Ok(None),
553            };
554
555            // Skip field filters if skip_fields is true
556            if skip_fields && filter_ctx.semantic_type() == SemanticType::Field {
557                continue;
558            }
559
560            // Get the column directly by its projected index
561            let column_idx = flat_format.projected_index_by_id(filter_ctx.column_id());
562            if let Some(idx) = column_idx {
563                let column = &input.columns()[idx];
564                let result = filter.evaluate_array(column).context(RecordBatchSnafu)?;
565                mask = mask.bitand(&result);
566            } else if filter_ctx.semantic_type() == SemanticType::Tag {
567                // Column not found in projection, it may be a tag column.
568                // Decodes primary keys if not already decoded.
569                if decoded_pks.is_none() {
570                    decoded_pks = Some(decode_primary_keys(self.codec.as_ref(), input)?);
571                }
572
573                let metadata = flat_format.metadata();
574                let column_id = filter_ctx.column_id();
575
576                // Check cache first
577                let tag_column = if let Some(cached_column) = decoded_tag_cache.get(&column_id) {
578                    cached_column.clone()
579                } else {
580                    // For dense encoding, we need pk_index. For sparse encoding, pk_index is None.
581                    let pk_index = if self.codec.encoding() == PrimaryKeyEncoding::Sparse {
582                        None
583                    } else {
584                        metadata.primary_key_index(column_id)
585                    };
586                    let column_index = metadata.column_index_by_id(column_id);
587
588                    if let (Some(column_index), Some(decoded)) =
589                        (column_index, decoded_pks.as_ref())
590                    {
591                        let column_metadata = &metadata.column_metadatas[column_index];
592                        let tag_column = decoded.get_tag_column(
593                            column_id,
594                            pk_index,
595                            &column_metadata.column_schema.data_type,
596                        )?;
597                        // Cache the decoded tag column
598                        decoded_tag_cache.insert(column_id, tag_column.clone());
599                        tag_column
600                    } else {
601                        continue;
602                    }
603                };
604
605                let result = filter
606                    .evaluate_array(&tag_column)
607                    .context(RecordBatchSnafu)?;
608                mask = mask.bitand(&result);
609            }
610            // Non-tag column not found in projection.
611        }
612
613        Ok(Some(mask))
614    }
